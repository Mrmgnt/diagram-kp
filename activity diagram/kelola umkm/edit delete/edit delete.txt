@startuml
title Activity Diagram: Edit dan Hapus Data UMKM (Ringkas)

|Admin/Pengguna Sistem|
start
:Diasumsikan pengguna sudah berada di halaman daftar UMKM;
:Memilih UMKM dari daftar untuk dikelola;

|System|
:Menampilkan opsi "Edit" dan "Hapus" untuk UMKM yang dipilih;

|Admin/Pengguna Sistem|
if (Pilih opsi "Edit UMKM"?) then (Ya)
    |System|
    :Menampilkan form edit dengan data UMKM terpilih;
   
    |Admin/Pengguna Sistem|
    :Mengubah data pada form;
    :Klik "Update";
    |System|
    if (Data valid?) then (Ya)
        :Memperbarui data UMKM di database;
        :Notifikasi "Data berhasil diperbarui";
    else (Tidak)
        :Notifikasi "Error validasi";
        :Form edit tetap ditampilkan dengan pesan error;
    endif
else if (Pilih opsi "Hapus UMKM"?) then (Ya)
    |System|
    :Menampilkan dialog konfirmasi "Yakin hapus data ini?";
    |Admin/Pengguna Sistem|
    if (Konfirmasi "Ya"?) then (Ya)
        |System|
        :Menghapus data UMKM dari database;
        :Notifikasi "Data berhasil dihapus";
    else (Tidak / Batal)
        :Proses dibatalkan;
    endif
else (Tidak memilih aksi / Kembali)
    :Kembali ke daftar UMKM (atau tetap di halaman);
endif
stop

@enduml
