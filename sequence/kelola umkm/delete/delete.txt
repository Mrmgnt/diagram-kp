@startuml
!theme materia

title Sequence Diagram - Hapus Data UMKM (Pola MVC)
hide footbox

actor "Pengelola" as Pengguna
boundary "View" as View
control "Controller" as Controller
entity "Model" as Model

Pengguna -> View: Memilih UMKM & klik tombol "Hapus"
activate View

View --> Pengguna: Menampilkan dialog konfirmasi "Anda yakin?"

alt Konfirmasi "Ya"
    Pengguna -> View: Klik tombol konfirmasi "Ya"
    
    View -> Controller: Minta hapus data UMKM (berdasarkan ID)
    activate Controller

    Controller -> Model: Panggil fungsi hapusDataUMKM(ID)
    activate Model
    
    Model -> Model: Menghapus data UMKM dari database
    Model --> Controller: Konfirmasi data telah dihapus
    deactivate Model
    
    Controller --> View: Respon sukses
    deactivate Controller
    
    View --> Pengguna: Notifikasi "Data berhasil dihapus" & perbarui daftar
    deactivate View
else Konfirmasi "Tidak / Batal"
    Pengguna -> View: Klik tombol "Tidak / Batal"
    
    View --> Pengguna: Menutup dialog konfirmasi
    deactivate View
end


@enduml
