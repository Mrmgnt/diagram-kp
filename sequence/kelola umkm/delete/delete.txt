@startuml
!theme materia

title Sequence Diagram - Hapus Data UMKM (Pola MVC)
hide footbox

actor "Pengelola" as Pengguna
boundary "Halaman_kelola_UMKM" as Halaman_kelola_UMKM
control "Controller" as Controller
entity "Entity" as Entity

Pengguna -> Halaman_kelola_UMKM: Memilih UMKM & klik tombol "Hapus"
activate Halaman_kelola_UMKM

Halaman_kelola_UMKM --> Pengguna: Menampilkan dialog konfirmasi "Anda yakin?"

alt Konfirmasi "Ya"
    Pengguna -> Halaman_kelola_UMKM: Klik tombol konfirmasi "Ya"
    
    Halaman_kelola_UMKM -> Controller: Minta hapus data UMKM (berdasarkan ID)
    activate Controller

    Controller -> Entity: Panggil fungsi hapusDataUMKM(ID)
    activate Entity
    
    Entity -> Entity: Menghapus data UMKM dari database
    Entity --> Controller: Konfirmasi data telah dihapus
    deactivate Entity
    
    Controller --> Halaman_kelola_UMKM: Respon sukses
    deactivate Controller
    
    Halaman_kelola_UMKM --> Pengguna: Notifikasi "Data berhasil dihapus" & perbarui daftar
    deactivate Halaman_kelola_UMKM
else Konfirmasi "Tidak / Batal"
    Pengguna -> Halaman_kelola_UMKM: Klik tombol "Tidak / Batal"
    
    Halaman_kelola_UMKM --> Pengguna: Menutup dialog konfirmasi
    deactivate Halaman_kelola_UMKM
end


@enduml
